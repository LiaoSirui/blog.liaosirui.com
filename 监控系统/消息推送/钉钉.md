发送信息示例

```python
"""Send message to dingtalk."""

import base64
import hashlib
import hmac
import time
from urllib.parse import quote

import httpx
from rich import print

# 自定义机器人 Webhook
DINGTALK_URL = "https://oapi.dingtalk.com/robot/send?access_token=xxxx"
# 自定义机器人加签密钥
DINGTALK_SECRET = "secret"


def dingtalk_sign():
    """加签."""
    # 获取当前时间戳，并将其转换为毫秒级
    timestamp = int(time.time() * 1000)
    # 将时间戳和钉钉应用的密钥拼接在一起，将拼接后的字符串转换为字节数组
    sign_before = f"{timestamp}\n{DINGTALK_SECRET}"
    # 用 HMAC-SHA256算法对字节数组进行签名
    h_sha256 = hmac.new(DINGTALK_SECRET.encode("utf-8"), sign_before, hashlib.sha256)
    # 将签名进行 Base64 编码，将编码后的签名进行 URL 编码
    sign = quote(base64.b64encode(h_sha256.digest()))
    return {"timestamp": timestamp, "sign": sign}


def send_message(
    title: str,
    message: str,
    at_mobiles: list[str] | None = None,
    is_at_all: bool = False,
):
    """发送消息.

    每个机器人每分钟最多发送 20 条消息到群里, 如果超过 20 条, 会限流 10 分钟

    - 发送消息不 `@` 任何人: `send_message(title="异常告警", message="服务器检测到异常！")`
    - 发送消息并 `@` 指定人员: `send_message(title="异常告警", message="服务器检测到异常！", at_mobiles=["13888888888"])`
    - 发送消息并 `@` 所有人员: `send_message(title="异常告警", message="服务器检测到异常！", is_at_all=True)`
    """
    if at_mobiles is None:
        at_mobiles = []
    print("发送内容:", message, at_mobiles, is_at_all)
    json = {
        "msgtype": "text",
        "text": {
            "content": message,
        },
        "markdown": {
            "title": "杭州天气",
            "text": message,
        },
        "at": {"atMobiles": at_mobiles, "isAtAll": is_at_all},
    }
    sign = dingtalk_sign()
    response = httpx.post(url=DINGTALK_URL, params=sign, json=json)
    print("响应内容:", response.json())


if __name__ == "__main__":
    send_message(title="异常告警", message="服务器检测到异常！", is_at_all=True)

```
